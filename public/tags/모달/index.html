모달

Parallel Routes로 구현한 모달을 Overlay-kit를 활용해서 구현한 모달로 바꾼 이유

React에서 모달을 구현하는 방법 React에서 모달을 어떻게 구현하는지 먼저 살펴봐야 할 것 같아요. 기본적인 구현 import React from "react"; interface ModalProps { isOpen: boolean; onClose: () => void; children: React.ReactNode; } const Modal = ({ isOpen, onClose, children }: ModalProps) => { if (!isOpen) { return null; } return ( <div className="fixed inset-0 z-[1000] flex items-center justify-center bg-black/50" onClick={onClose} > <div className="z-[1001] min-w-[300px] max-w-[90%] rounded-lg bg-white p-5 shadow-lg" onClick={(e) => e.stopPropagation()} > {children} <button onClick={onClose} className="mt-4 rounded bg-gray-200 px-4 py-2 hover:bg-gray-300"> 닫기 </button> </div> </div> ); }; export default Modal;
